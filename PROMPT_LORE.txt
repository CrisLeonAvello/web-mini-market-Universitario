Resumen de la conversación sobre checkout.html y su integración:

1. Se solicitó el código para la página checkout.html, compatible con el diseño y lógica del proyecto StudiMarket.
2. Se generó un checkout.html moderno, con resumen de productos, formulario de usuario y botón de pago. El código incluye estilos, renderizado dinámico de productos desde el carrito (localStorage) y confirmación de compra.
3. Se explicó cómo enlazar el checkout desde index.html, modificando el botón "Pagar" del carrito para redirigir a checkout.html.
4. Se agregó el evento en main.js para que el botón de pago redirija correctamente a checkout.html:

    payBtn.onclick = function() {
        window.location.href = 'checkout.html';
    };

5. Se confirmó que el flujo de compra está integrado y funcional.

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Checkout - StudiMarket</title>
    <link rel="stylesheet" href="styles.css">
    <style>
        .checkout-container { max-width: 700px; margin: 40px auto; background: #fff; border-radius: 16px; box-shadow: 0 4px 16px rgba(0,0,0,0.08); padding: 32px; }
        .checkout-header { text-align: center; margin-bottom: 32px; }
        .checkout-summary { margin-bottom: 24px; }
        .checkout-summary h3 { color: #4285f4; margin-bottom: 12px; }
        .checkout-products { margin-bottom: 24px; }
        .checkout-product { display: flex; align-items: center; margin-bottom: 12px; padding-bottom: 12px; border-bottom: 1px solid #eee; }
        .checkout-product img { width: 60px; border-radius: 8px; margin-right: 16px; }

        .checkout-form label { display: block; margin-bottom: 6px; color: #333; }

        .checkout-btn { width: 100%; padding: 14px; background: #007bff; color: #fff; border: none; border-radius: 10px; font-size: 18px; cursor: pointer; margin-top: 12px; }
        .checkout-btn:hover { background: #0056b3; }
    </style>
</head>
<body>
    <div class="checkout-container">
        <div class="checkout-header">
            <h1>Finalizar Compra</h1>
            <p>Revisa tu pedido y completa tus datos para pagar</p>
        </div>
        <div class="checkout-summary">
            <h3>Resumen de tu carrito</h3>
            <div class="checkout-products" id="checkout-products">
                <!-- Los productos se insertan aquí con JS -->
            </div>
            <div class="checkout-total" id="checkout-total">
                <!-- Total se inserta aquí -->
            </div>
        </div>
        <form class="checkout-form" id="checkout-form">
            <label for="name">Nombre completo</label>
            <input type="text" id="name" name="name" required>
            <label for="email">Correo electrónico</label>
            <input type="email" id="email" name="email" required>
            <label for="address">Dirección de entrega</label>
            <input type="text" id="address" name="address" required>
            <label for="payment">Método de pago</label>
            <select id="payment" name="payment" required>
                <option value="">Selecciona...</option>
                <option value="tarjeta">Tarjeta de crédito/débito</option>
                <option value="transferencia">Transferencia bancaria</option>
                <option value="efectivo">Pago en efectivo</option>
            </select>
            <button type="submit" class="checkout-btn">Confirmar y Pagar</button>
        </form>
    </div>
    <script>
        // Obtener productos del carrito desde localStorage
        function getCart() {
            return JSON.parse(localStorage.getItem('cart') || '[]');
        }
        // Obtener productos globales
        async function getProducts() {
            const res = await fetch('products.json');
            return await res.json();
        }
        // Renderizar productos en el checkout
        async function renderCheckoutProducts() {
            const cart = getCart();
            const products = await getProducts();
            const container = document.getElementById('checkout-products');
            const totalDiv = document.getElementById('checkout-total');
            let total = 0;
            container.innerHTML = '';
            cart.forEach(item => {
                const product = products.find(p => p.id === item.id);
                if (product) {
                    const subtotal = product.precio * item.qty;
                    total += subtotal;
                    container.innerHTML += `
                        <div class="checkout-product">
                            <img src="${product.imagenes[0]}" alt="${product.titulo}">
                            <div class="checkout-product-info">
                                <div class="checkout-product-title">${product.titulo}</div>
                                <div class="checkout-product-qty">Cantidad: ${item.qty}</div>
                            </div>
                            <div class="checkout-product-price">$${subtotal.toLocaleString()}</div>
                        </div>
                    `;
                }
            });
            totalDiv.textContent = `Total a pagar: $${total.toLocaleString()}`;
        }
        // Enviar formulario
        document.getElementById('checkout-form').addEventListener('submit', function(e) {
            e.preventDefault();
            alert('¡Pago realizado con éxito! Gracias por tu compra.');
            localStorage.removeItem('cart');
            window.location.href = 'index.html';
        });
        // Inicializar
        renderCheckoutProducts();
    </script>
</body>
</html>
